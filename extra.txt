js for photo frame:



 const photoUrls = [
      "photo1.jpg",
      "photo2.jpg",
      // Add more photo URLs here
    ];

    let currentPhotoIndex = 0;

    const photoElement = document.getElementById("photo");
    const customPreviousBtn = document.getElementById("customPreviousBtn");
    const customNextBtn = document.getElementById("customNextBtn");

    function updatePhoto() {
      photoElement.src = photoUrls[currentPhotoIndex];
    }

    function previousPhoto() {
      currentPhotoIndex = (currentPhotoIndex - 1 + photoUrls.length) % photoUrls.length;
      updatePhoto();
    }

    function nextPhoto() {
      currentPhotoIndex = (currentPhotoIndex + 1) % photoUrls.length;
      updatePhoto();
    }

    // Use your custom button IDs for event handling
    customPreviousBtn.addEventListener("click", previousPhoto);
    customNextBtn.addEventListener("click", nextPhoto);

    updatePhoto();
